function ErrorCode:OnErrorRep(err_code, ...)
    local args = {...}
    Network:ClearWait()
    Network:CacheMsgClear()
    if err_code ~= ErrorCode.SUCCESS then
        Chapter:ClearAutoData()
    end
    if err_code == ErrorCode.SUCCESS then
        -- 无返回协议，清除网络状态
    elseif err_code == ErrorCode.ERROR_SHOP_REPEAT_REQ then
        
    elseif err_code == ErrorCode.ERROR_DUNGEON_SETTLING then
        Groble_UIMgr:ShowMessageBox(true, true, LanguageMgr:GetTextByID(120023),
                function()
                    uimgr:onMainScene()
                end, nil, true)
    elseif err_code == ErrorCode.ERROR_VERSION then
        local errmsg = ""
        for i, v in ipairs(args) do
            errmsg = errmsg .. " " .. v
        end
        Groble_UIMgr:ShowMessageBox(true, false, errmsg)
    elseif err_code == ErrorCode.ERROR_DUNGEON_SETTLING then
        Groble_UIMgr:ShowGameWordTipsLayer(8880014)
    elseif err_code == ErrorCode.SERVER_BUSY then
        Groble_UIMgr:ShowGameWordTipsLayer(8880015)
    elseif err_code == ErrorCode.ERROR_REQ_FAST then
        Groble_UIMgr:ShowGameWordTipsLayer(8880016)
    elseif err_code == ErrorCode.ERROR_NAME_UNLAWFUL then
        Groble_UIMgr:ShowGameWordTipsLayer(8880017)
    elseif err_code == ErrorCode.ERROR_COMMENT_MSGLEN_REACH_MAX then
        Groble_UIMgr:ShowGameWordTipsLayer(8880017)
    elseif err_code == ErrorCode.ERROR_CHARGE_ERROR_NTF then
        local errmsg = ""
        if tostring(args[1]) == "1" then
            errmsg = LanguageMgr:GetTextByID(8880018)
        elseif tostring(args[1]) == "2" then
            errmsg = LanguageMgr:GetTextByID(8880019)
        elseif tostring(args[1]) == "3" then
            errmsg = LanguageMgr:GetTextByID(8880020)
        end
        Groble_UIMgr:ShowMsgLayer(errmsg, function()end, false, true)
    elseif err_code == ErrorCode.ERROR_CHARGE_NUM_MAX then
        Groble_UIMgr:ShowGameWordTipsLayer(8880021)
    elseif err_code == ErrorCode.ERROR_REVED_AUTH_GIFT then
        UnIndulge:GiftFireEvent()
    elseif err_code == ErrorCode.ERROR_PFRIEND_PF_FULL then
        Groble_UIMgr:ShowGameWordTipsLayer(8880025)
    elseif err_code == ErrorCode.ERROR_MAIL_GOT_UNIT then
        Groble_UIMgr:ShowGameWordTipsLayer(8880026)
    elseif err_code == ErrorCode.ERROR_COMMENT_MSG_ID_NOT_EXIST then
        --评论被删除
        Groble_UIMgr:ShowGameWordTipsLayer(ErrorCode.ERROR_COMMENT_MSG_ID_NOT_EXIST)
        Comment:RemoveComment();

    elseif err_code == ErrorCode.ERROR_RANK_SETTLE_NIL then
        -- 找不到历史排行数据
    elseif err_code == ErrorCode.ERROR_CHARGE_WG_TRIGGER_ING then
        -- 礼包正在触发
    elseif err_code == ErrorCode.ERROR_CHARGE_WG_TRIGGER_COUNT_MAX then
        
    elseif err_code == ErrorCode.ERROR_WALL_LOCK then
        Groble_UIMgr:ShowGameWordTipsLayer(880084)
    --调查问卷错误领取 
    elseif err_code == ErrorCode.ERROR_LEVEL_NO_GAMING then
        -- 关卡已结算
        Groble_UIMgr:ShowMessageBox(true, true, "关卡已结算，奖励已获取",
            function()
                uimgr:onMainScene()
            end, nil, true)
    elseif err_code == ErrorCode.ERROR_URL_GIFT_NO_CFG or err_code == ErrorCode.ERROR_URL_GIFT_GOT then
        --啥也不干
    else
    --print("---------------------------------err code : ", err_code, " err msg : ", table.unpack(args))
        local errmsg = "net error code: " .. err_code .. " msg: "
        for i, v in ipairs(args) do
            errmsg = errmsg .. " " .. v
        end
        Groble_UIMgr:ShowMessageBox(true, false, errmsg)
    end
end
